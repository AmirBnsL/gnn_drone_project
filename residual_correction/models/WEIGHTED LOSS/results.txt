def compute_loss(pred, target):
    # residual magnitude
    mag = torch.norm(target, dim=1)

    # smooth weights instead of binary jump
    weights = 1.0 + 2.0 * mag  # linear scaling

    mse = (pred - target) ** 2
    mse = mse.mean(dim=1)

    return (weights * mse).mean()
