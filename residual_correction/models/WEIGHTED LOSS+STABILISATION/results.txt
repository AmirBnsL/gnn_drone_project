def compute_loss(pred, target):
    mag = torch.norm(target, dim=1)

    # normalize magnitude to avoid extreme weights
    mag = mag / (mag.mean() + 1e-6)

    weights = 1.0 + mag  # softer scaling

    mse = (pred - target) ** 2
    mse = mse.mean(dim=1)

    return (weights * mse).mean()
